السلام عليكم ورحمة الله وبركاته أولا أطلب من الإدارة الكريمة عدم إغلاق الموضوع .. أو حذفه .. لأهميته والموضوع باختصار .. أنني أنا ( معتز بتوحيده ) أبو بصير الطرطوسي !!!! هذا خبر موجز وإليكموه بالتفصيل : قام أحد أعضاء التنديد ( بنت حماس في الإخلاص ) .. فادعى أنني العضو الكافر بالطاغوت وأنني لا أمانع في تكفير الشيخ البنا والشيخ أحمد ياسين رحمهما الله .. كل هذا ليثبت للناس أنني تكفيري .. لينقذ بقرته المقدسة من مقصلة بيان القاعدة .. ولكن لم يخبرنا : هل أخو من أطاع الله والسويلمي تكفيريان ؟ المهم .. قوله أنني الكافر بالطاغوت أضحوكة تدل على غبائه .. لأنه استدل بالتشابه بين مشاركتنا وقد كانت بيني وبين الكافر بالطاغوت مناظرة علمية عن طريق الخاص في الإصلاح احتفظت بها .. لأني أعتز بالمناظرات العلمية .. كان هو يرى أن جهاد المقدسي فك الله أسره أعظم من جهاد بن لادن .. فراجعته برفق ولين لأني أعرف صفاء منهجه وإخلاصه أحسبه والله حسيبه لكن عمر حنيف ( على المصري ) طار بها وكتب موضوعا بعنوان : التنكيل بما قال الكافر بالطاغوت من أباطيل وجعلها توقيعا له .. نكاية في الكافر بالطاغوت لأنه يعتقده أبا بصير وهذه صور للمراسلات الخاصة بين الكافر بالطاغوت ومعتز بتوحيده : وهذا رد معتز على الرسالة : وهذه رسالة أخرى : وهذه رد عليها : وهذه ثالثة : والرد عليها من معتز : إذا ليس معتز هو الكافر بالطاغوت كما قالت بنت حماس .. أرأيتم كذبا ودجلا مثل ذلك قال أكفر أحمد ياسين والبنا قال !!!! كل ذلك ليبرر للناس كفريات عجله المقدس التي أظهرتُها ووضحتُها حتى لم يعد أحد يخفى عليه حاله خاصة بعد بيان القاعدة القاتل لقد تركوا الرد الشرعي على كفريات عجلهم والتفتوا للقيل والقال .. فلان هو فلان .. والسب والشتم والقذارة والنتن .. لماذا لاتردوا عن كفريات المسعور بدلا من هذا الدجل ؟ المهم .. نأتي لعمر حنيف إذ يتدخل بأحد أشهر معرفاته ( المحضار ) قائلا : وبما أنني عندهم الكافر بالطاغوت .. فأنا ابو بصير .. إذا أنا أشارك بينكم .. فرحبوا بي ( ابتسامة ) وهم يشنون حملة على أبي بصير عن طريق ( عمر حنيف ) بمعرفاته المتعددة .. لا أدري لماذا ؟ ما الهدف من إسقاط أبي بصير .. ؟ ما هذه الحملة الشعواء على الرجل ؟ لا تتعجبوا لأنه تجرأ ونقد العجل المقدس .. فقالوا عنه : أبو بعير !!!! لكن ربما يكون عند عمر حنيف أسباب أخرى .. الله أعلم !!!!! هو الآن يحاول أن يثبت أنه لا صلة له بالمعرفات التي أشرنا إليها فيجعل المعرفات تشارك مشاركات متتالية في موضوع واحد وكل معرف له دور .. ويعلن بأحد المعرفات أنه سيعتزل النت ( ابتسامة ) .. ويؤثر السلامة رغم ان المعرف الذي يبغي الاعتزال ويتظاهر بالهدوء .. كان شرسا جدا في الإصلاح ( على المصرى) مثل عمر حنيف بالضبط .. مما تسبب في طرده !!!! كان جل همه إسقاط المقدسي !!! مرة أبو بصير .. ومرة المقدسي !! ماذا يريد ؟ لا أدري .. أخيرا .. فقد راسلت الإدارة بشأن الموضوع الذي يقولون إن البراق أنزله عن رؤوس القبوريين .. فأخبروني بالحقائق التالية : اقتباس: أولا : الذي أنزل الموضوع ( الزين القابسي ) تبين من خلال الدعم الفني بالشبكة أنه قبوري عضروط ( مشهور جدا لديهم ) .. وتم طرده في الحال .. وحذف الموضوع ثانيا : حذف الموضوع في زمن قياسي .. رغم ذلك تمكن من أنزله من تصويره فورا لنقله لمزبلة المنتديات .. ليتهموا البراق أنها تتبنى هذا التخريف .. وهذا يدل أن الموضوع أحبك وخطط له في أروقة عجل اسكوتانديارد .. لكن خاب مسعاهم .. ثالثا : حاول العضروط القبوري التسجيل مرة ثالثة .. لأنه مطرود مرتين من قبل .. وها هو واقف على باب الدعم الفني يأكل حذاءه يطلب التفعيل !!! ليحاول أن يقوم بعمل تمثيلية قبورية جديدة في سيرك التهريج القبوري لكن هيهات .. هذا كل ما يستطيعون فعله .. الدجل .. والسب واللعن .. هيا يا إخوة التوحيد .. رحبوا بأخيكم أبي بصير الطرطوسي .. !!! الذي هو معتز بتوحيده !!! أريد ترحيبا حارا .. هل بعد ذلك تصدقوا لهم حرفا ؟ يريدون هدم صروح الجهاد إنقاذا لمسعور ضال .. قتله المجاهدون ببيانهم لكن أنى لهم ذلك .. وقد فشل رامسفيلد في المهمة .. صحيح أنهم حملوها نيابة عنه .. لكننا نعاهد الله أن نفشل خطتهم كما أفشلنا خطة رامسفيلد فهما ( القبوريون وامسفيلد ) في بوتقة واحدة وهدفهم واحد اللهم رد كيدهم في نحورهم .. اللهم عليك بهم فإنهم لا يعجزونك اللهم انصر المجاهدين في كل مكان .. وآخر دعوانا أن الحمد لله رب العالمين وصلى الله وسلم وبارك على سيدنا محمد وعلى آله وصحبه وسلم ..

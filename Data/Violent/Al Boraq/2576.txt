الـكـنـدي قـال : كـنـتُ جـالـسـاً عـنـد رسـولِ الله صـلـى اللهُ عـلـيـه وسـلـمَ فـقـال رجـل : يـا رسـول الله أذال الـنَّاسُ الـخـيـلَ ، ووضـعـوا الـسـلاحَ ، وقـالـوا : لا جـهـادَ قـد وضَـعـت الـحـربُ أوزَارَهـا ، فـأقـبـلَ رسـول الله صـلـى الله عـلـيـه وسـلـم بـوجـهـهِ وقـال : كـذَبـوا ، الآن الآن جـاء الـقـتـال ، ولا يـزالُ مـن أمـتـي أمـةٌ يُـقـاتـلـون .... الـحـديـث (62) ، فـهـذا يُـبـيـنُ أن الـجـهـاد الـمـقـصـود بِـه الـقـتـال الـمـعـروف ، وهـذا كـاف لـصـفـعِ هـؤلاءِ ، وكـشـف زيـفِـهِـم ، ولـكـن قـاتـلَ الله أهـل الأهـواء ؛ وإذا قُـلـنـا هـم أهـلُ الـحـديـثِ بـالـمـصـطـلـحِ قـلـت : فـإنـهـم كـانـوا أربـابُ جـهـادٍ فـي كُـلِّ الـمـيـاديـن ، فـلـذلـك يـقـول الـشـيـخ سـلـيـمـان بـن مـحـمـد بـن عـبـد الـوهـاب &quot; إذ يـمـتـنـع أن تـكـون الـطـائـفـة الـمـنـصـورة لا تـعـرف الـحـديـث ولا سـنـن رسـول الله صـلـى الله عـلـيـه وسـلـم ، بـل لا يـكـون مـنـصـوراً عـلـى الـحـقِ إلا مـن عـمـلَ بـكـتـابِ الله وسـنـةِ رسـوله صـلـى الله عـلـيـه وسـلـم ، وهـم أهـلُ الـحـديـثِ مـن الـعـربِ وغـيـرِهـم &quot; (63) . *..*..**..**..**..**..**..**..**..**..** *..*الـوصـيـة *..* *..*..**..**..**..**..**..**..**..**..** قـال ابـن الـقـيـمِ رحـمـه الله تـعـالـى كـلامـاً غـايـةً فـي *..*..**..**..**..**..**..**..**..**..** *..*خـاتـمـة *..* *..*..**..**..**..**..**..**..**..**..** فـلا يُـفـهـم مـن هـذا أن الـطَّائـفـةَ الـمـنـصـورة لـيـس لهـا مُـهـمـة إلا الـقـتـال بـالـسـنـانِ فـحـسـب ، بـل الـجـهـاد ، جـهـاد قـرآنٍ وسـنـان ، وهـل الـجـهـاد إلا دعـوةٌ للـحـقِ ، أو الـدفـاع عـنـه ؛ الـطـائـفـةُ هـي الـتـي سـارت حـيـث سـار الـنـبـي صـلـى الله عـلـيـه وسـلـم ، ووقـفـت حـيـثُ وقـف ، وقـاتـلـت حـيـث قـاتـل ، وانـتـهـت حـيـث انـتـهـى ، ووُجِـدَت حـيـثُ أَمـرَ ، هـي وارثـةُ الهُـدى ، وقـامـعـةُ الهـوى ، ولا يـتـعـارض هـذا مـع قـول أهـل الـعـلـمِ الـذيـن قـالـوا بـأنـهـم أهـل الـحـديـث كـمـا ذكـرنـا آنـفـاً . ولا أنسى أن أشكر جميع إخواني الأفاضل الذين سألوا عني في حال مرضي وراسلوني وأعتذر عن عدم الرد لانشغالي الكبير بالمرض والذي أسأل الله أن يكشفه عني وعن كل مصاب إنه جواد
